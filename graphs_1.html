Dept First search : Iterative 

        const dfs = (graph, source) => {
          let stack = [ source ];
          while (stack.length > 0) {
            const cur = stack.pop();
            console.log(cur);
            for (var neighbore of graph[cur]) {
              stack.push(neighbore);
            }
          }
        }

Dept First search : recursive 

        const deptFirstSearch = (graph, source)=> {
          console.log(source);
          for(let neighbore of graph[source]) {
            deptFirstSearch(graph, neighbore);
          }
        }
